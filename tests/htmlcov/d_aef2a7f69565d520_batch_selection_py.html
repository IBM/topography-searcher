1""" Module containing batch selection methods for Bayesian optimisation. 

2 The methods take in a network encoding of the surface topography, 

3 and analyse it to select a diverse set of minima. """

13def get_excluded_minima(ktn: type, 

14 energy_cutoff: float = 1.0, 

15 penalise_edge: bool = False, 

16 coords: type = None, 

17 penalise_similarity: bool = False, 

18 proximity_measure: float = 0.2, 

19 known_points: NDArray = None) -> list: 

20 """ Apply specified criteria to decide if the minima should 

21 be allowed in the BayesOpt batch. Returns the list of those 

22 that should be excluded """ 

23 

24 energies = get_minima_energies(ktn) 

25 # Get the cutoffs for allowed minima 

26 abs_energy_cutoff = np.min(energies) + \ 

27 energy_cutoff*(np.max(energies) - np.min(energies)) 

28 excluded_minima = get_minima_above_cutoff(ktn, abs_energy_cutoff) 

29 if penalise_edge: 

30 edge_minima = get_bounds_minima(ktn, coords) 

31 excluded_minima += edge_minima 

32 if penalise_similarity: 

33 similar_minima = get_similar_minima(ktn, proximity_measure, 

34 known_points) 

35 excluded_minima += similar_minima 

36 # Retain unique elements 

37 return list(set(excluded_minima))

40def select_batch(ktn: type, batch_size: int, batch_selection_method: str, 

41 fixed_batch_size: bool, barrier_cutoff: float = 1.0, 

42 excluded_minima: list = None) -> tuple[list, NDArray]: 

43 

44 """ 

45 Method to select the batch of points from a given network 

46 Returns the indices of the minima and their coordinates 

47 Type of batch selection depends on batch_selection_method 

48 Lowest - Select the batch_size lowest minima in value 

49 Monotonic - Select monotonic sequence basins of the network 

50 Barrier - Select minima that are separated by sufficiently large 

51 barriers counting from lowest in acquisition 

52 Topographical - First perform Monotonic, then fill batch using Barrier 

53 """ 

54 

55 if excluded_minima is None: 

56 excluded_minima = [] 

57 # Get the value of acquisition function for each minimum 

58 energies = get_minima_energies(ktn) 

59 # Get the cutoffs for allowed minima 

60 abs_barrier_cutoff = barrier_cutoff * \ 

61 (np.max(energies) - np.min(energies)) 

62 # Write the batch scheme we are using 

63 with open('logfile', 'a', encoding="utf-8") as outfile: 

64 outfile.write("------- BATCH SELECTION ---------\n") 

65 outfile.write(f"{batch_selection_method}: {batch_size} minima\n") 

66 # Perform the batch selection scheme 

67 batch_indices = generate_batch(ktn, batch_selection_method, 

68 excluded_minima, 

69 abs_barrier_cutoff) 

70 # If we want a fixed batch size fill with lowest minima not in batch 

71 if fixed_batch_size and (len(batch_indices) < batch_size): 

72 batch_indices = fill_batch(ktn, batch_indices, excluded_minima) 

73 # Limit the batch to batch_size if more have been returned 

74 if np.size(batch_indices) > batch_size: 

75 batch_indices = batch_indices[:batch_size] 

76 batch_points = get_batch_positions(ktn, batch_indices) 

77 return batch_indices, batch_points

80def generate_batch(ktn: type, batch_selection_method: str, 

81 excluded_minima: list, 

82 absolute_barrier_cutoff: float) -> NDArray: 

83 """ Generate the batch given the specified method """ 

84 if batch_selection_method == "Lowest": 

85 batch_indices = lowest_batch_selector(ktn, excluded_minima) 

86 elif batch_selection_method == "Monotonic": 

87 batch_indices = monotonic_batch_selector(ktn, excluded_minima) 

88 elif batch_selection_method == "Barrier": 

89 batch_indices = barrier_batch_selector(ktn, excluded_minima, 

90 absolute_barrier_cutoff) 

91 elif batch_selection_method == "Topographical": 

92 batch_indices = topographical_batch_selector(ktn, excluded_minima, 

93 absolute_barrier_cutoff) 

94 return batch_indices

97def lowest_batch_selector(ktn: type, excluded_minima: list) -> list: 

98 """ Simply select the lowest minima of the acquisition function 

99 Returns list of indices the selected minima correspond to """ 

100 # Starting from the lowest indices progressively add if not banned 

101 ordered_indices = get_ordered_minima(ktn).tolist() 

102 batch_indices = [i for i in ordered_indices if i not in excluded_minima] 

103 return batch_indices

106def fill_batch(ktn: type, batch_indices: list, 

107 excluded_minima: list) -> NDArray: 

108 """ Return the batch after adding the remaining 

109 lowest minima in order """ 

110 # Use lowest_batch_selector to fill in with lowest minima 

111 lowest_indices = lowest_batch_selector(ktn, excluded_minima+batch_indices) 

112 # And combine the two batches 

113 batch_indices += lowest_indices 

114 return batch_indices

117def topographical_batch_selector(ktn: type, 

118 excluded_minima: list, 

119 absolute_barrier_cutoff: float) -> NDArray: 

120 """ Select a batch of minima based on the surface topography 

121 First finds monotonic sequence basins, and if not batch_size 

122 of them then use barrier_batch_selector to generate remaining 

123 Returns the indices of the selected nodes, and their 

124 corresponding coordinates """ 

125 

126 # Get the monotonic batch 

127 monotonic_indices = monotonic_batch_selector(ktn, excluded_minima) 

128 with open('logfile', 'a', encoding="utf-8") as outfile: 

129 outfile.write(f"Got {np.size(monotonic_indices)} " 

130 f"minima from monotonic sequence basins\n") 

131 # Get the barrier batch 

132 barrier_indices = barrier_batch_selector(ktn, 

133 excluded_minima, 

134 absolute_barrier_cutoff, 

135 monotonic_indices) 

136 barrier_indices = \ 

137 [i for i in barrier_indices if i not in monotonic_indices] 

138 return monotonic_indices + barrier_indices

141def barrier_batch_selector(ktn: type, 

142 excluded_minima: list, 

143 absolute_barrier_cutoff: float, 

144 current_batch_indices: list = None): 

145 """ Select a batch of points that are all separated by barrier heights 

146 greater than absolute_barrier_cutoff. Returns the indices """ 

147 

148 if current_batch_indices is None: 

149 current_batch_indices = [] 

150 batch_indices = [] 

151 # Find maximum ts energy for barrier calc 

152 ts_energies = [] 

153 for u, v in ktn.G.edges(): 

154 ts_energies.append(ktn.get_ts_energy(u, v)) 

155 if not ts_energies: 

156 max_ts_energy = 1e5 

157 else: 

158 max_ts_energy = np.max(ts_energies) 

159 # Find energy range for barrier calc 

160 energies = get_minima_energies(ktn) 

161 e_range = np.max(energies) - np.min(energies) 

162 # Loop over all minima in network 

163 ordered_minima_indices = get_ordered_minima(ktn) 

164 for i in ordered_minima_indices: 

165 # Minimum is already in batch or banned so ignore 

166 if (i in current_batch_indices) or (i in excluded_minima): 

167 continue 

168 # Loop over current batch and check if allowed by barrier heights 

169 allowed = True 

170 for j in current_batch_indices: 

171 accept = sufficient_barrier(ktn, i, j, max_ts_energy, e_range, 

172 absolute_barrier_cutoff) 

173 # If conditions fail compared to any of current batch - reject 

174 if not accept: 

175 allowed = False 

176 # If the minimum has passed all checks then add to batch 

177 if allowed: 

178 # Add to batches 

179 batch_indices.append(i) 

180 current_batch_indices.append(i) 

181 return batch_indices

184def sufficient_barrier(ktn: type, node_i: int, node_j: int, 

185 max_ts_energy: float, e_range: float, 

186 absolute_barrier_cutoff: float) -> bool: 

187 """ Check if minima pass criteria to be added to the batch 

188 Returns logical specified if pass or fail """ 

189 height = disconnected_height(ktn, node_i, node_j, max_ts_energy, e_range) 

190 if height > 1e9: 

191 return False 

192 # Barriers, calculated in either way, from both minima 

193 barrier1 = height - ktn.get_minimum_energy(node_i) 

194 barrier2 = height - ktn.get_minimum_energy(node_j) 

195 # If barrier less than minimum allowed then reject 

196 if min(barrier1, barrier2) < absolute_barrier_cutoff: 

197 return False 

198 return True

201def monotonic_batch_selector(ktn: type, excluded_minima: list) -> list: 

202 """ Find all monotonic sequence basins when excluding excluded_minima, 

203 and return their indices in increasing function value """ 

204 

205 batch_indices = [] 

206 # Loop from lowest to highest energy 

207 ordered_indices = get_ordered_minima(ktn) 

208 for i in ordered_indices: 

209 monotonic = True 

210 energy_i = ktn.get_minimum_energy(i) 

211 edges_i = ktn.G.edges(i) 

212 # If no edges then ignore the minimum 

213 if (not edges_i) or (i in excluded_minima): 

214 continue 

215 # Loop over all edges getting the two connected minima 

216 for j in edges_i: 

217 if j[0] == j[1]: 

218 continue 

219 if j[1] == i: 

220 j.reverse() 

221 min1, min2 = j 

222 if min2 in excluded_minima: 

223 continue 

224 # Find other connected minimum and if lower not monotonic 

225 if ktn.get_minimum_energy(min2) <= energy_i: 

226 monotonic = False 

227 # If not connected to any lower minima then add to MSB 

228 if monotonic: 

229 batch_indices.append(i) 

230 return batch_indices

233def get_batch_positions(ktn: type, batch_indices: list) -> NDArray: 

234 """ Get the corresponding batch position from the list of indices """ 

235 batch_points = np.zeros((len(batch_indices), 

236 ktn.get_minimum_coords(0).size), dtype=float) 

237 for idx, i in enumerate(batch_indices, 0): 

238 batch_points[idx, :] = ktn.get_minimum_coords(i) 

239 return batch_points

242def evaluate_batch(true_potential: type, batch_indices: list, 

243 batch_points: NDArray) -> NDArray: 

244 """ Evaluate the true potential at each point in the given batch 

245 Return the training and corresponding response values """ 

246 new_response = np.apply_along_axis( 

247 true_potential.function, 1, batch_points) 

248 # Write the batch to file 

249 for i in range(len(batch_indices)): 

250 with open('logfile', 'a', encoding="utf-8") as outfile: 

251 outfile.write(f"{batch_indices[i]} Coords: ") 

252 for j in range(np.size(batch_points[i])): 

253 outfile.write(f"{batch_points[i][j]} ") 

254 outfile.write(f" Function: {new_response[i]}\n") 

255 return new_response
