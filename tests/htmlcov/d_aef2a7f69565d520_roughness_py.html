2 https://doi.org/10.26434/chemrxiv-2023-0zx26 """

8def roughness_metric(ktn: type, lengthscale: float = 0.8) -> float: 

9 """ Compute the roughness metric for the current kinetic 

10 transition network """ 

11 lengthscale = lengthscale**2 

12 # Roughness is zero for one minimum or empty network 

13 if ktn.n_minima in (0, 1): 

14 return 0.0 

15 frustration = 0.0 

16 # Loop over all minima in the set 

17 for i in range(ktn.n_minima): 

18 # Get the minimum energy 

19 min_energy = ktn.get_minimum_energy(i) 

20 # Find the directly connected transition states 

21 connections = get_connections(ktn, i) 

22 # Compute the barrier to each of these transition states 

23 for ts in connections: 

24 ts_energy = ktn.get_ts_energy(i, ts) 

25 barrier = ts_energy - min_energy 

26 if ts_energy < min_energy: 

27 barrier = 0.0 

28 population = get_population(ktn, i, ts, lengthscale) 

29 frustration += (population*barrier) 

30 frustration /= ktn.n_minima 

31 return frustration

34def get_population(ktn: type, min_node: int, ts_node: int, 

35 lengthscale: float) -> float: 

36 """ Compute the population of a given minimum in the network. 

37 Population is the value of the RBF kernel at the separation 

38 between minimum and transition state """ 

39 min_coords = ktn.get_minimum_coords(min_node) 

40 ts_coords = ktn.get_ts_coords(min_node, ts_node) 

41 dist = np.linalg.norm(min_coords - ts_coords) 

42 population = np.exp(-1.0*((dist**2)/(2.0*lengthscale))) 

43 return population
